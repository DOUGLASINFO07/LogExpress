2º) Usar o script para criar a tabela contatos
CREATE TABLE Contatos( 
`id` int(10) unsigned NOT NULL auto_increment, 
`nome` varchar(50) default NULL, 
`email` varchar(100) default NULL, 
  PRIMARY KEY (`id`) 
);
